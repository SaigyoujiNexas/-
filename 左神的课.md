## 只出现一次的数

其他数字都出现偶数次

```c
int onlyOneNum(int num[], int n)
{
    int eor = 0;
    for(int i = 0; i < n; i++)
    {
        eor ^= num[i];
    }
    return eor;
}
```



## 两个只出现一次的数

其余数字均出现两次

```c
int* singleNumber(int* nums, int numsSize, int* returnSize){
    long eor = 0;
    long eor2 = 0;
    int * ret =(int *)malloc(sizeof(int) * 2);
    for(int i = 0; i < numsSize; i++)
        eor ^= nums[i];		
    //now, eor = a ^ b; 
        //eor一定不为零
    //if any position is zero, then must be 0 ^ 1
    long rightOne = eor & (~eor + 1);		//获取最右边的一位
    for(int i = 0; i < numsSize; i++)
    {
        if(nums[i] & rightOne)
            eor2 ^= nums[i];
    }
    eor ^= eor2;
    ret[0] = eor;
    ret[1] = eor2;
    *returnSize = 2;
    return ret;
}
```

# 排序算法

## 插入排序

```c
void insertSort(int* nums, int n)
{
	int j, key;
	if (n == 0 || n == 1)
		return;

	for (int i = 1; i < n; i++)
	{
		key = nums[i];
		for (j = i - 1; j >= 0 && nums[j] > key; j--)
			nums[j + 1] = nums[j];
		nums[j + 1] = key;
	}
}
```

## 二分搜索

在一个有序数组找该数的存在。

```c
int binarySearch(int * arr, int size_n, int num){
    int left = 0;
    int right = size_n - 1;
    int middle = ((right - left) >> 1) + left;
    while(left < right && arr[middle] != num){
        middle = ((right - left) >> 1) + left;
        if(arr[middle] > num)
            right = middle;
        else if(arr[middle] < num)
            left = middle;
        else
            return middle;
    }
    return -1;
}

```

用处：

* 找某个数是否在有序数组中存在
* 找>=某个数最左边的值
* 局部最小

```c
int getMax(int*arr, int size){
    return process(arr, 0, size - 1);
}
int process(int* arr, int l, int r){
    if(l == r) return arr[l];
    int middle = ((r - l) >> 1) + l;
    int right = process(arr, l, middle);
    int left = process(arr, middle + 1, r);
    return max(arr[middle], right, left);
}
int max(int a, int b, int c){
    int max;
    max = a > b? a: b;
    max = max > c? max: c;
    return max;
}
```

对于递归行为有master公式：

T(N) = a * (T(n/b)) + O(n ^ d)

在这里如上的getMax， T(N) = 2 * T(N / 2) + O(1)

时间复杂度计算如下

logb(a) > d -> O(N ^ log(b, a))

logb(a) = d -> O(N ^ d * logN)

log(b, a) < d -> O(N ^ d)

```c
    if(left == right) return;
```

```c
    process(arr, left, middle, cnt);

    int * help = malloc(sizeof(int) * (right - left + 1));
```
